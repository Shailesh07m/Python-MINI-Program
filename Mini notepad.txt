filename = "mynotes.txt"  # File the program will use

while True:  # Loop so menu keeps showing until user exits
    print("\n===== MINI NOTEPAD =====")
    print("1. Write to file (overwrite)")
    print("2. Read file")
    print("3. Append to file")
    print("4. Exit")

    choice = input("Enter your choice (1-4): ")

    # OPTION 1 — Write (old content deleted)
    if choice == "1":
        text = input("Enter text to write into file: ")
        try:
            with open(filename, "w") as file:
                file.write(text + "\n")  # Write new text
            print("File written successfully.")
        except Exception as e:
            print("Error:", e)

    # OPTION 2 — Read file
    elif choice == "2":
        try:
            with open(filename, "r") as file:
                content = file.read()
                print("\n--- File Content ---")
                print(content)
        except FileNotFoundError:
            print("File does not exist yet.")
        except Exception as e:
            print("Error:", e)

    # OPTION 3 — Append (add without deleting)
    elif choice == "3":
        text = input("Enter text to append: ")
        try:
            with open(filename, "a") as file:
                file.write(text + "\n")  # Add text at end
            print("Text appended successfully.")
        except Exception as e:
            print("Error:", e)

    # OPTION 4 — Exit program
    elif choice == "4":
        print("Exiting Mini Notepad. Goodbye!")
        break  # Stops the loop

    else:
        print("Invalid choice. Please select 1–4.")
